    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const questions = [
                "Would you rather be able to talk to animals or be able to speak every human language fluently?",
                "Would you rather live without music or live without movies?",
                "Would you rather have unlimited access to the world's best food or the world's best books?",
                "Would you rather be an Olympic gold medalist or a Nobel Prize winner?",
                "Would you rather have a remote control that could pause time or a button that could teleport you instantly?",
                "Would you rather fight 100 duck-sized horses or one horse-sized duck?",
                "Would you rather know the date of your death or the cause of your death?",
                "Would you rather be famous and disliked, or unknown and loved?",
                "Would you rather be permanently stuck as a child or permanently stuck as an elderly person?",
                "Would you rather live in a world where you could fly, but only as fast as you can walk, or teleport, but only 10 feet at a time?",
                "Would you rather have hands that constantly feel wet or feet that constantly smell?",
                "Would you rather have an unlimited supply of batteries or an unlimited supply of petrol/gasoline?",
                "Would you rather travel to the past or travel to the future?"
            ];

            const questionText = document.getElementById('question-text');
            const generateBtn = document.getElementById('generate-btn');
            const copyBtn = document.getElementById('copy-btn');
            const copyConfirmation = document.getElementById('copy-confirmation');

            // Function to get a random question
            const generateQuestion = () => {
                const randomIndex = Math.floor(Math.random() * questions.length);
                questionText.textContent = questions[randomIndex];
            };

            // Function to copy the question to clipboard
            const copyQuestion = () => {
                // Use the textContent of the display element
                const textToCopy = questionText.textContent;
                
                // Attempt to copy
                navigator.clipboard.writeText(textToCopy)
                    .then(() => {
                        // Show confirmation message
                        copyConfirmation.style.opacity = '1';
                        // Hide it after 2 seconds
                        setTimeout(() => {
                            copyConfirmation.style.opacity = '0';
                        }, 2000);
                    })
                    .catch(err => {
                        console.error('Failed to copy text: ', err);
                        // Fallback message for older browsers or restricted environments (though unlikely with modern API)
                        questionText.textContent = "Error copying text. Please select and copy manually.";
                    });
            };

            // Event Listeners
            generateBtn.addEventListener('click', generateQuestion);
            copyBtn.addEventListener('click', copyQuestion);

            // Generate a starting question
            generateQuestion();
        });
    </script>
